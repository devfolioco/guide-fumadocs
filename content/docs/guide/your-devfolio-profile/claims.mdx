title: Claims

---

import { Tabs, Tab } from "fumadocs-ui/components/tabs";


<Tabs items={["Schwag", "Bounties", "Onchain Credentials"]} defaultValue="Schwag">
  <Tab value="Schwag">
    1. Click on your profile in the top-right corner and choose 'My Claims' or click [here](https://devfolio.co/claims)  
    <img src="/assets/image 135 (1) (1).png" alt="My Claims" />

    2. Select the Schwag card.  
    <img src="/assets/image 133 (1).png" alt="Schwag Card" />

    3. Click 'Claim schwag' for the initiative you want.  
    <img src="/assets/pika-1682066192886-1x 1.png" alt="Claim Schwag" />

    4. Fill out the necessary details in the form and submit it.  
    <img src="/assets/pika-1682066306949-1x 1.png" alt="Fill Schwag Form" />

    5. Woohoo! Your schwag will be sent to you soon!  
    <img src="/assets/image (18).png" alt="Schwag Confirmation" />
  </Tab>

  <Tab value="Bounties">
    1. Click on your profile in the top-right corner and choose 'My Claims.'  
    <img src="/assets/image 135 (1).png" alt="My Claims for Bounties" />

    2. Select the Bounties card on the Claims page.  
    <img src="/assets/pika-1682066146228-1x 2.png" alt="Bounties Card" />

    3. Click 'Claim bounties' for the initiative you want.  
    <img src="/assets/pika-1682066768151-1x 1.png" alt="Claim Bounty" />

    4. Start the KYC process.  
    <img src="/assets/Screenshot (3236).png" alt="Start KYC" />

    5. Verify your Identity.  
    <img src="/assets/Screenshot (3237).png" alt="Verify Identity" />

    6. Select Document "Country" and "Type".  
    <img src="/assets/Screenshot (3239).png" alt="Select Country and Type" />

    7. Submit the documents and wait for confirmation.  
    <img src="/assets/Screenshot (3245).png" alt="Submit Docs" />

    8. Click 'Confirm Wallet' to complete the payment.  
    <img src="/assets/pika-1682066851084-1x 1 (1).png" alt="Confirm Wallet" />

    9. Yay! Your bounties will be sent to you soon!  
    <img src="/assets/pika-1682073331112-1x 1.png" alt="Bounty Confirmation" />
  </Tab>

  <Tab value="Onchain Credentials">
    1. Click on your profile in the top-right corner and choose 'My Claims' or click [here](https://devfolio.co/claims)  
    <img src="/assets/image 135 (1) (2).png" alt="My Claims Onchain" />

    2. Select the Onchain credentials card.  
    <img src="/assets/image 133.png" alt="Onchain Credential Card" />

    3. Choose the credential that you want to claim.  
    <img src="/assets/image 137 (3) (1) (1).png" alt="Choose Credential" />

    4. Link your Ethereum wallet to Devfolio.  
    <img src="/assets/image 139.png" alt="Link Wallet" />

    5. Click on Continue with Ethereum and finish the transaction.  
    <img src="/assets/image 146 (1).png" alt="Continue with Ethereum" />

    6. Double-check the wallet details and click on ‘Confirm Wallet’.  
    For gasless Onchain Credentials the transaction is executed by Devfolio.  
    <img src="/assets/image (1) (1) (1) (1) (1) (1) (1).png" alt="Confirm Wallet" />

    7. Hooray! The SBT will be sent to your connected wallet.  
    You can view it on OpenSea & EAS Scan and tweet about it!  
    <img src="/assets/image (2) (1) (1) (1).png" alt="SBT Confirmation" />
  </Tab>
</Tabs>
